\begin{definition}[Even natural number]
\label{def:even}
A natural number $n$ is even if there exists a natural number $k$ such that $n = 2k$.
\end{definition}

\begin{lemma}[Adding two preserves evenness]
\label{lem:plus-two}
If $n$ is even, then $n + 2$ is even.
\end{lemma}

\begin{proof}
Suppose $n$ is even. Then there exists a natural number $k$ such that $n = 2k$.
Hence
\[
n + 2 = 2k + 2 = 2(k+1),
\]
so $n + 2$ is even.
\end{proof}

\begin{theorem}[Sum of two even numbers]
\label{thm:sum-even}
If $m$ and $n$ are even, then $m+n$ is even.
\end{theorem}

\begin{proof}
Since $m$ is even, there exists a natural number $a$ such that $m = 2a$.
Since $n$ is even, there exists a natural number $b$ such that $n = 2b$.
Therefore
\[
m+n = 2a + 2b = 2(a+b),
\]
so $m+n$ is even.
\end{proof}
